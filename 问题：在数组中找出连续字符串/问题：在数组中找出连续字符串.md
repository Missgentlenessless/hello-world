问题：在数组中找出连续字符串

```
f([{0: '产品经理asdadadsa'},{1: '产品'},{2: '产品经'},{3: '理'},{4: '经'}, {5: '理'}, {6: 'ads'},{7: '产'},{8: '品经'},{9: '理'},], '产品经理')
因输出[0,2,7]
```

```
 我自己写的方法：
 function f(arr,str) {
                let index=[]
                let number=0
                let newarr=[]
                let bool=false
                let length =str.length
                arr.forEach((item,ind)=>{
                    if(item[ind].length>=length&&item[ind].includes(str)){
                            index.push(ind)
                            bool=true
                    }else{
                        if( str[0]===item[ind][0] ){
                            newarr=[item[ind]]
                            bool?bool=false:index.pop();
                            index.push(ind)
                            number=item[ind].length
                        }else if(str.includes(item[ind])&&str[number]===item[ind][0] ){
                            newarr.push(item[ind])
                            number+=item[ind].length
                        }else{
                            newarr=[]
                            number=0
                        }
                        if( newarr.join('').length>=length&&newarr.join('').includes(str) ){
                            newarr=[]
                            number=0
                            bool=true
                        }
                    }
                })
                return index
            }
```
